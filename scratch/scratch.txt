SELECT t.name, c.name, c.column_index, d.name, category.name
FROM dbmodel.database_model_table t
JOIN dbmodel.database_model_column as c ON t.id = c.table_model_id
JOIN dbmodel.database_model_data_type as d ON d.id = c.data_type_id
JOIN dbmodel.database_model_column_category as category ON category.id = c.column_category_id
order by t.name, c.column_index asc;
